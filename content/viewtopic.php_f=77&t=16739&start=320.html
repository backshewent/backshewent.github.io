(
  lv_idbasket bb_basketitem.idbasket%TYPE
)
RETURN NUMBER
IS
  lv_pounds_ordered NUMBER(36,2);
BEGIN
  SELECT SUM(lbs)
  INTO lv_pounds_ordered
  FROM (
    SELECT CASE
      WHEN OPTION1 = 1 THEN QUANTITY * 0.5
      WHEN OPTION1 = 2 THEN QUANTITY
      ELSE 0 END AS lbs
    FROM bb_basketitem

  );
RETURN lv_pounds_ordered;
END;
